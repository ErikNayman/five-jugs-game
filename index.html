<script>

let health = 100;

let happiness = 90;

let savings = 1000;

let investments = 1000;

let activeIncome = 2000;

let passiveIncome = 50;

let turn = 0;

let maxTurns = 10;


function startGame() {

  document.getElementById('start-screen').classList.add('hidden');

  document.getElementById('game-screen').classList.remove('hidden');

  updateBars();

}


function nextTurn() {

  turn++;

  maxTurns = document.getElementById('turns').value;


  if (turn > maxTurns || health <= 0 || happiness <= 0) {

    alert("Игра окончена! Вы прошли " + (turn-1) + " ходов.");

    location.reload();

    return;

  }


  const income = activeIncome + passiveIncome;

  const expenses = 1300;

  let freeMoney = income - expenses;

  document.getElementById('freeMoney').innerText = freeMoney;


  // Монте-Карло рост инвестиций

  const investChange = Math.random() <= 0.7 ? Math.random()*0.10 : Math.random()*-0.05;

  investments *= (1 + investChange);


  // Случайные события

  const eventRoll = Math.random();

  if (eventRoll < 0.15) {

    savings -= 100;

    alert("Непредвиденные траты: -$100");

  } else if (eventRoll < 0.25) {

    investments *= 0.9;

    alert("Мини-кризис: инвестиции упали на 10%");

  } else if (eventRoll < 0.30) {

    savings += 500;

    alert("Премия! +$500 к сбережениям");

  } else if (eventRoll < 0.36) {

    health -= 10;

    alert("Болезнь: здоровье -10%");

  } else if (eventRoll < 0.40) {

    activeIncome *= 1.05;

    alert("Повышение на работе! Доход +5%");

  }


  // Пассивное восстановление здоровья при хорошем состоянии

  if (health < 100 && happiness > 70) {

    health += 3;

    if (health > 100) health = 100;

  }


  // Если переработка — падение счастья

  if (freeMoney < 0) {

    happiness -= 5;

  }


  // Обновление шкал

  updateBars();

}


function updateBars() {

  document.getElementById('healthBar').style.width = health + '%';

  document.getElementById('happinessBar').style.width = happiness + '%';

  document.getElementById('savingsBar').style.width = Math.min((savings/10000)*100, 100) + '%';

  document.getElementById('investmentBar').style.width = Math.min((investments/10000)*100, 100) + '%';

}
