<script>

let turn = 0;

let maxTurns = 0;

let savings = 0;

let investments = 0;

let health = 100;

let happiness = 90;

let scores = [];


// график Chart.js

const ctx = document.getElementById('scoreChart').getContext('2d');

const chart = new Chart(ctx, {

  type: 'line',

  data: {

    labels: [],

    datasets: [{

      label: 'Накопительный результат (баллы)',

      data: [],

      borderColor: 'rgb(75, 192, 192)',

      tension: 0.3,

      fill: false

    }]

  },

  options: {

    scales: { y: { beginAtZero: true } }

  }

});


function startGame(turns) {

  maxTurns = turns;

  document.getElementById('maxTurns').innerText = turns;

  document.getElementById('start-screen').classList.add('hidden');

  document.getElementById('game-screen').classList.remove('hidden');

  updateBars();

}


function runTurn() {

  turn++;

  const survival = parseFloat(document.getElementById('survival').value);

  const comfort = parseFloat(document.getElementById('comfort').value);

  const save = parseFloat(document.getElementById('save').value);

  const invest = parseFloat(document.getElementById('invest').value);

  const healthSpend = parseFloat(document.getElementById('healthSpend').value);

  const edu = parseFloat(document.getElementById('edu').value);

  const extra = document.getElementById('extra').value;


  let income = 1000;

  if (extra === "да") income += 200;


  let event = "ничего";

  let expensePenalty = 0;

  let crisisLoss = 0;

  let healthPenalty = 0;


  const random = Math.random();

  if (random < 0.10) {

    event = "мелкая трата –$50";

    expensePenalty = -50;

  } else if (random < 0.20) {

    event = "мини-кризис: –6% инвестиций";

    crisisLoss = investments * 0.06;

  } else if (random < 0.25) {

    event = "болезнь –5% здоровья";

    healthPenalty = health * 0.05;

  }


  const totalSpent = survival + comfort + save + invest + healthSpend + edu;

  const free = income - totalSpent + expensePenalty;


  savings += save + free;

  const growth = Math.random() <= 0.7 ? Math.random()*0.03 : Math.random()*-0.01;

  investments = (investments + invest - crisisLoss) * (1 + growth);


  health -= healthPenalty;

  if (extra === "да") health -= health * 0.05;

  if (survival < 500) health -= ((500 - survival) / 500) * 100;

  if (survival >= 500) health += 500 / 100;

  if (happiness >= 90) health += 500 / 100;

  health += (healthSpend / 500) * 50;

  health = Math.max(0, Math.min(100, Math.round(health)));


  if (comfort < 500) happiness -= ((500 - comfort) / 500) * 10;

  if (extra === "да") happiness -= 0.01 * 500;

  happiness = Math.max(0, Math.round(happiness));


  updateBars();

  updateSummary(event, free);

  updateHistory(event, free);

  updateChart();


  if (health <= 0 || happiness <= 0 || turn >= maxTurns) {

    let reason = "Ты завершил все ходы.";

    if (health <= 0) reason = "Ты потерял здоровье. Игра окончена.";

    if (happiness <= 0) reason = "Ты потерял счастье. Игра окончена.";

    finishGame(reason);

  }

}


function updateBars() {

  document.getElementById('healthBar').style.width = `${health}%`;

  document.getElementById('happinessBar').style.width = `${happiness}%`;

}


function updateSummary(event, free) {

  document.getElementById('turn').innerText = turn;

  document.getElementById('event').innerText = event;

  document.getElementById('free').innerText = `$${Math.round(free)}`;

  document.getElementById('savings').innerText = `$${Math.round(savings)}`;

  document.getElementById('investments').innerText = `$${Math.round(investments)}`;

  document.getElementById('healthResult').innerText = `${health}%`;

  document.getElementById('happinessResult').innerText = `${happiness}%`;

}


function updateHistory(event, free) {

  const table = document.getElementById("historyTable");

  const row = document.createElement("tr");

  row.innerHTML = `

    <td class="p-2">${turn}</td>

    <td class="p-2">${event}</td>

    <td class="p-2">$${Math.round(free)}</td>

    <td class="p-2">$${Math.round(savings)}</td>

    <td class="p-2">$${Math.round(investments)}</td>

    <td class="p-2">${health}%</td>

    <td class="p-2">${happiness}%</td>

  `;

  table.appendChild(row);

}


function updateChart() {

  let score = 0;

  score += Math.floor(savings / 1000);

  score += Math.floor(investments / 1000) * 2;

  if (health > 70) score += 20;

  if (happiness > 70) score += 20;

  const passiveIncome = (investments * 0.05 + savings * 0.03) / 12;

  if (passiveIncome > 1000) score += 30;


  scores.push(score);

  chart.data.labels.push(turn);

  chart.data.datasets[0].data.push(score);

  chart.update();

}


function finishGame(reasonText) {

  document.getElementById('game-screen').classList.add('hidden');

  document.getElementById('end-screen').classList.remove('hidden');

  document.getElementById('endReason').innerText = reasonText;


  const passiveIncome = (investments * 0.05 + savings * 0.03) / 12;

  let finalScore = scores[scores.length - 1];

  let title = "Новичок";

  if (finalScore >= 80) title = "Финансовый магнат";

  else if (health > 90) title = "Здоровяк";

  else if (happiness > 90) title = "Счастливчик";

  else if (health < 30) title = "Выживший";


  document.getElementById('finalSavings').innerText = `$${Math.round(savings)}`;

  document.getElementById('finalInvestments').innerText = `$${Math.round(investments)}`;

  document.getElementById('finalHealth').innerText = `${Math.round(health)}%`;

  document.getElementById('finalHappiness').innerText = `${Math.round(happiness)}%`;

  document.getElementById('finalPassiveIncome').innerText = `$${Math.round(passiveIncome)}`;

  document.getElementById('finalScore').innerText = `${finalScore} баллов`;

  document.getElementById('finalTitle').innerText = title;

}
